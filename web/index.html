Language Features

Explicit Control

No hidden magic. All variables and parameters have explicit types. Structured control flow with if, while, for, and switch statements. Compile to efficient x86-64 assembly.

Memory Management

Leverage the built-in conservative garbage collection via new T for heap allocation, or manage it yourself with C malloc when you need precise control.

Seamless FFI

First-class C interoperability. Declare extern function and extern var to effortlessly interface with external libraries like Winsock2 and the MSVC runtime.

Module System

Organize your codebase efficiently. Import standard libraries or your own modules to keep namespaces clean and dependency resolution robust.

Performance

Compiles directly to x86-64 NASM assembly. No runtime overhead beyond what you explicitly use. Perfect for systems programming and performance-critical applications.

Rich Type System

Comprehensive primitive types (int8-int64, uint8-uint64, float32/64), structs, enums, pointers, arrays, and strings with proper memory layout for C interop.

import "std/io";
import "std/math";

function factorial(n: int32) -> int32 {
  if (n <= 1) {
    return 1;
  }
  return n * factorial(n - 1);
}

function main() -> int32 {
  var result: int32 = factorial(10);
  println_int(result);
  return 0;
}
struct Point {
  x: float64;
  y: float64;
  
  function distance(other: Point*) -> float64 {
    var dx: float64 = this.x - other->x;
    var dy: float64 = this.y - other->y;
    return sqrt(dx*dx + dy*dy);
  }
}

function main() -> int32 {
  var p1: Point = {0.0, 0.0};
  var p2: Point = {3.0, 4.0};
  var dist: float64 = p1.distance(&p2);
  return 0;
}
import "std/net";
import "std/io";

function main() -> int32 {
  if (net_init() != 0) {
    puts("Failed to initialize network");
    return 1;
  }
  
  var sock: int64 = socket_tcp();
  var addr: cstring = sockaddr_in_any(8080);
  
  if (bind(sock, addr, 16) != 0) {
    puts("Bind failed");
    return 1;
  }
  
  listen(sock, 5);
  puts("Server listening on port 8080");
  
  // Accept loop...
  return 0;
}
